```jsx
// Required in order to not have the default message of "Less than two mfa types available"
const MFATypeOptions = {
  SMS: true,
  Optional: true,
  TOTP: true,
};

<amplify-select-mfa-type MFATypes={MFATypeOptions}></amplify-select-mfa-type>;
```
